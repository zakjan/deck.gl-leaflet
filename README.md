
or

```
<script src="https://unpkg.com/deck.gl-leaflet@1.3.1/dist/deck.gl-leaflet.min.js"></script>
```

## Usage

```
import { LeafletLayer } from 'deck.gl-leaflet';
```

```
const map = L.map(...);
const deckLayer = new LeafletLayer({
  views: [
    new MapView({
      repeat: true
    })
  ],
  layers: [
    new ScatterplotLayer({
      data: [...],
    })
  ]
});
map.addLayer(deckLayer);
```
